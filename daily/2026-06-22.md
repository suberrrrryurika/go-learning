## 💻 代码示例

### 示例 1: 基础示例

```go
package main

import (
    "context"
    "fmt"
    "sync"
    "time"
)

// WorkerPool 实现
type WorkerPool struct {
    workers   int
    taskQueue chan func()
    wg        sync.WaitGroup
}

func NewWorkerPool(workers int) *WorkerPool {
    wp := &WorkerPool{
        workers:   workers,
        taskQueue: make(chan func(), 100),
    }
    wp.start()
    return wp
}

func (wp *WorkerPool) start() {
    for i := 0; i < wp.workers; i++ {
        wp.wg.Add(1)
        go func(id int) {
            defer wp.wg.Done()
            for task := range wp.taskQueue {
                task()
            }
        }(i)
    }
}

func (wp *WorkerPool) Submit(task func()) {
    wp.taskQueue <- task
}

func (wp *WorkerPool) Shutdown() {
    close(wp.taskQueue)
    wp.wg.Wait()
}

func main() {
    pool := NewWorkerPool(4)
    
    // 提交任务
    for i := 0; i < 10; i++ {
        taskID := i
        pool.Submit(func() {
            fmt.Printf("Worker 处理任务 %d\n", taskID)
            time.Sleep(100 * time.Millisecond)
        })
    }
    
    time.Sleep(500 * time.Millisecond)
    pool.Shutdown()
    fmt.Println("所有任务完成")
}
```

### 示例 2: 进阶示例

```go
package main

import (
    "fmt"
    "runtime"
    "runtime/pprof"
    "os"
)

// 性能分析示例
func profileDemo() {
    // CPU 分析
    f, _ := os.Create("cpu.prof")
    pprof.StartCPUProfile(f)
    defer pprof.StopCPUProfile()
    
    // 内存分析
    runtime.GC()
    pprof.WriteHeapProfile(os.Stdout)
    
    // 显示 goroutine 数量
    fmt.Printf("Goroutines: %d\n", runtime.NumGoroutine())
}
```
